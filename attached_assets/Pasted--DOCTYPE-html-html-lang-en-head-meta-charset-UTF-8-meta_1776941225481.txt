function getUnlockDate(day) {
  const d = new Date(BIRTHDAY);
  d.setDate(d.getDate() - (5 - day));
  d.setHours(0, 0, 0, 0);
  return d;
}

function getCurrentDay() {
  if (DEV_DAY !== null) return DEV_DAY;
  const now = new Date();
  // Walk backwards from day 5 to find which day is "today"
  for (let d = 5; d >= 1; d--) {
    const unlock = getUnlockDate(d);
    const next   = d < 5 ? getUnlockDate(d + 1) : new Date(BIRTHDAY.getTime() + 86400000);
    if (now >= unlock && now < next) return d;
  }
  // Before May 25 = 0 (nothing unlocked yet)
  // After birthday = 6 (show post-birthday state)
  const now2 = new Date();
  if (now2 >= getUnlockDate(5)) return 5; // keep birthday accessible on birthday + after
  return 0;
}

function launchConfetti() {
  const cv = document.getElementById('confetti-canvas');
  cv.style.display = 'block';
  cv.width = window.innerWidth; cv.height = window.innerHeight;
  const ctx = cv.getContext('2d');
  const cols = ['#f5e6e0','#e8b4a0','#c47a6a','#c9956c','#9e6b72','#fdf0ec','#fff9f5','#f9c6b5','#ffd6cc'];
  const pieces = Array.from({ length: 160 }, () => ({
    x: Math.random()*cv.width, y: -20 - Math.random()*120,
    r: 3 + Math.random()*8, d: 0.6 + Math.random()*2.2,
    dx: (Math.random()-0.5)*0.8,
    color: cols[~~(Math.random()*cols.length)],
    ta: 0, ts: 0.04 + Math.random()*0.08,
    shape: Math.random() > 0.4 ? 'rect' : 'circle',
  }));
  let f = 0;
  (function draw() {
    ctx.clearRect(0, 0, cv.width, cv.height);
    pieces.forEach(p => {
      p.ta += p.ts; p.y += p.d; p.x += p.dx;
      if (p.y > cv.height+20) { p.y = -10; p.x = Math.random()*cv.width; }
      ctx.fillStyle = p.color;
      ctx.save(); ctx.translate(p.x, p.y); ctx.rotate(p.ta);
      if (p.shape === 'circle') { ctx.beginPath(); ctx.arc(0,0,p.r,0,Math.PI*2); ctx.fill(); }
      else { ctx.fillRect(-p.r, -p.r*.5, p.r*2, p.r); }
      ctx.restore();
    });
    if (++f < 520) requestAnimationFrame(draw);
    else { ctx.clearRect(0, 0, cv.width, cv.height); cv.style.display = 'none'; }
  })();
}
